// --- FULL REPLACEMENT CORE MATRIX ENGINE FOR OMEGA v300++ --- async function buildDiscreteEngine() { if (isEngineBuilt) return; // 1. INPUT SPLITTING (True Dual Channel) splitterNode = audioCtx.createChannelSplitter(2); mergerNode = audioCtx.createChannelMerger(2); // 2. INITIALIZE ANALYSERS analyserL = audioCtx.createAnalyser(); analyserL.fftSize = 512; analyserR = audioCtx.createAnalyser(); analyserR.fftSize = 512; // 3. FULL DISCRETE PATH FOR L & R (Independent Processing) // Kita buat 2 Array untuk semua nodes agar matrix bisa dikontrol via loop const channels = ['l', 'r']; const nodes = {}; channels.forEach(ch => { // High-Precision Filter Stage nodes[`hpf${ch.toUpperCase()}`] = audioCtx.createBiquadFilter(); nodes[`lpf${ch.toUpperCase()}`] = audioCtx.createBiquadFilter(); // Harmonic & Sultan Logic nodes[`sultan${ch.toUpperCase()}`] = audioCtx.createBiquadFilter(); nodes[`harmonic${ch.toUpperCase()}`] = audioCtx.createBiquadFilter(); // Dynamic Gain Stage nodes[`gain${ch.toUpperCase()}`] = audioCtx.createGain(); // Multi-Band Limiter (Anti-Clip) nodes[`limiter${ch.toUpperCase()}`] = audioCtx.createDynamicsCompressor(); // Setup initial values nodes[`hpf${ch.toUpperCase()}`].type = "highpass"; nodes[`lpf${ch.toUpperCase()}`].type = "lowpass"; nodes[`limiter${ch.toUpperCase()}`].threshold.value = -3; nodes[`limiter${ch.toUpperCase()}`].knee.value = 0; nodes[`limiter${ch.toUpperCase()}`].ratio.value = 20; }); // 4. MAPPING NODE PATH (WIRING THE OMEGA MATRIX) // L-CHANNEL PATH streamNode.connect(splitterNode, 0, 0); splitterNode.connect(nodes.HPFL, 0); nodes.HPFL.connect(nodes.LPFL); nodes.LPFL.connect(nodes.SULTANL); nodes.SULTANL.connect(nodes.HARMONICL); nodes.HARMONICL.connect(nodes.GAINL); nodes.GAINL.connect(nodes.LIMITERL); nodes.LIMITERL.connect(analyserL); analyserL.connect(mergerNode, 0, 0); // R-CHANNEL PATH streamNode.connect(splitterNode, 0, 1); splitterNode.connect(nodes.HPFR, 1); nodes.HPFR.connect(nodes.LPFR); nodes.LPFR.connect(nodes.SULTANR); nodes.SULTANR.connect(nodes.HARMONICR); nodes.HARMONICR.connect(nodes.GAINR); nodes.GAINR.connect(nodes.LIMITERR); nodes.LIMITERR.connect(analyserR); analyserR.connect(mergerNode, 0, 1); mergerNode.connect(audioCtx.destination); // Assign to Global Scope Object.assign(window, nodes); isEngineBuilt = true; console.log("OMEGA v300++ MATRIX FIRMWARE LOADED - ALL SLOTS ACTIVE"); renderSystemLoop(); // Start visualizer }
